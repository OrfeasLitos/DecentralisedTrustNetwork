\begin{definition}[Total Incoming/Outgoing Trust]
\label{inouttrust}
  \begin{equation}
    in_{A, j} = \sum\limits_{v \in N^{-}\left(A\right)_j}DTr_{v \rightarrow A, j} \enspace, \:\:\:\:\:\:\:\:\:\:
    out_{A, j} = \sum\limits_{v \in N^{+}\left(A\right)_j}DTr_{A \rightarrow v, j}
  \end{equation}
\end{definition}
