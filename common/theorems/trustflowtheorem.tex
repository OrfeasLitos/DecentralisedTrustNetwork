\begin{theorem}[Trust Flow Theorem] \ \\
  \label{trustflow}
  Let $\mathcal{G}$ be a game graph and $A, B \in \mathcal{V}$. It holds that
  \begin{equation*}
    Tr_{A \rightarrow B} = maxFlow\left(A, B\right) \enspace.
  \end{equation*}
\end{theorem}
