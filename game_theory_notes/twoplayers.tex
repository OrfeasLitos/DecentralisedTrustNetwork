\begin{equation*}
      \begin{blockarray}{r@{}*{5}{c}}
       &           &   from   & \\
       &  DTr(B,A) & DTr(A,B) & Cap(A) \\
	\begin{block}{r[*{3}{c}]cc}
                       & x_1 & x_2 & x_3 & Purchase & \\
        Action(A) = {} & y_1 & 0   & y_3 & DTr(A,B) & to \\
                       & z_1 & z_2 & 0   & Cap(A)   & \\
	\end{block}

    % Purchase -|
    % DTr(A,B)  | to
    % Cap(A)   _|

\begin{align*}
  DTr'(A,B) &= DTr(A,B) + x_1 + z_1 - y_1 - y_3 \\
  DTr'(B,A) &= DTr(B,A) - x_1 - x_2 - x_3 \\
  Cap'(A)   &= Cap(A)   + x_2 + y_1 - z_1 - z_2 \\
  Bought    &= \frac{x_3 + y_3 + z_2}{cost(b)}
\end{align*}

\begin{gather*}
  DTr'(A,B) + DTr'(B,A) + Cap(A) + Bought \times cost(b) \\ = \\
  DTr(A,B) + DTr(B,A) + Cap(A)
\end{gather*}

\begin{align*}
  x_1 + x_2 + x_3 & \leq DTr(B,A) \\
  y_1 + y_3 & \leq DTr(A,B) \\
  z_1 + z_2 & \leq Cap(A)
\end{align*}
