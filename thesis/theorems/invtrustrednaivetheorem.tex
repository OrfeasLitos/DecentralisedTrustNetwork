\begin{theorem}[Invariable Trust Reduction with Naive Algorithms] \ \\
  \label{invariability}
  If $\forall i \in [n], c_i' \leq x_i$, then $||\delta_i||_1$ and $||\Delta_i||_1$ are independent of $x'_i, c'_i$.
\end{theorem}
