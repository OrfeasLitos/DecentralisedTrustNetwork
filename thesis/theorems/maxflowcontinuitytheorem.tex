\addcontentsline{toc}{subsection}{maxFlow Continuity Theorem}
\begin{theorem}[maxFlow Continuity] \ \\
  \label{maxflowcont}
  Suppose a directed weighted graph with a source $A$, a drain $B$ and a capacity configuration $C \in \mathcal{C}$. Let
  also $p \in \mathbb{N} \cup \{\infty\}$. The function $maxFlow : \mathcal{C} \rightarrow \mathbb{R}^{+}$ is continuous
  with respect to the $||\cdot||_p$ norm.
\end{theorem}
