\begin{theoremgr}{Θεώρημα Εμπιστοσύνης -- Ροής}{trustflow} \ \\
  Έστω $\mathcal{G}$ ένας γράφος παιχνιδιού και $A, B \in \mathcal{V}$. Ισχύει ότι
  \begin{equation*}
    Tr_{A \rightarrow B} = maxFlow\left(A, B\right) \enspace.
  \end{equation*}
\end{theoremgr}
